==================== docker-compose.yml ====================

services:

  # --- PHP / Laravel Backend ---
  app:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: quizzard_app
    restart: unless-stopped
    working_dir: /var/www
    volumes:
      - ./backend:/var/www
    networks:
      - quizzard_network
    depends_on:
      - db

  # --- Nginx Web Server ---
  nginx:
    image: nginx:alpine
    container_name: quizzard_nginx
    restart: unless-stopped
    ports:
      - "8000:80"
    volumes:
      - ./backend:/var/www
      - ./nginx/default.conf:/etc/nginx/conf.d/default.conf
    networks:
      - quizzard_network
    depends_on:
      - app

  # --- MySQL Database ---
  db:
    image: mysql:8.0
    container_name: quizzard_db
    restart: unless-stopped
    environment:
      MYSQL_DATABASE: quizzard
      MYSQL_ROOT_PASSWORD: root
      MYSQL_USER: quizzard_user
      MYSQL_PASSWORD: quizzard_pass
    ports:
      - "3306:3306"
    volumes:
      - db_data:/var/lib/mysql
    networks:
      - quizzard_network

networks:
  quizzard_network:
    driver: bridge

volumes:
  db_data:
